#!/usr/bin/env bash
set -euo pipefail

# Update README.md on all remote branches except main/master
# Default mode is dry-run; use --apply to commit and push.

## Usage:
#   ./update-readme.sh
#   ./update-readme.sh --apply
#   ./update-readme.sh --apply --repo /c/ari-git

REPO_DIR="/c/ari-git"
FILE="README.md"
REMOTE="origin"
DRY_RUN=true
COMMIT_MSG="Update README.md [ci skip]"

while [[ $# -gt 0 ]]; do
  case "$1" in
    --apply)
      DRY_RUN=false
      shift
      ;;
    --dry-run)
      DRY_RUN=true
      shift
      ;;
    --repo)
      REPO_DIR="${2:?missing value for --repo}"
      shift 2
      ;;
    --remote)
      REMOTE="${2:?missing value for --remote}"
      shift 2
      ;;
    --message)
      COMMIT_MSG="${2:?missing value for --message}"
      shift 2
      ;;
    --file)
      FILE="${2:?missing value for --file}"
      shift 2
      ;;
    -h|--help)
      sed -n '1,24p' "$0"
      exit 0
      ;;
    *)
      echo "Unknown arg: $1" >&2
      exit 2
      ;;
  esac
done

cd "$REPO_DIR"
[[ -d .git ]] || { echo "Not a git repo: $REPO_DIR" >&2; exit 1; }

if ! git diff --quiet || ! git diff --cached --quiet; then
  echo "Working tree is not clean. Commit/stash first." >&2
  exit 1
fi

git fetch "$REMOTE" --prune
START_BRANCH="$(git rev-parse --abbrev-ref HEAD)"

# List all remote branches except main/master/HEAD
mapfile -t branches < <(
  git for-each-ref "refs/remotes/${REMOTE}" --format='%(refname:short)' \
    | sed "s#^${REMOTE}/##" \
    | grep -Ev '^(HEAD|main|master)$' || true
)

if [[ ${#branches[@]} -eq 0 ]]; then
  echo "No branches found under ${REMOTE} to update."
  exit 0
fi

for b in "${branches[@]}"; do
  echo "Processing ${b}..."
  git switch -C "$b" "${REMOTE}/${b}" >/dev/null || { echo "  Skipped: cannot switch to ${b}"; continue; }

  # Make sure file exists before staging
  if [[ ! -f "$FILE" ]]; then
    echo "  Skipped: $FILE not found in branch $b"
    continue
  fi

  # Touch the file to trigger an update if needed
  git add "$FILE"

  if git diff --cached --quiet; then
    echo "  No change"
    continue
  fi

  if [[ "$DRY_RUN" == true ]]; then
    echo "  [DRY RUN] Would commit/push to ${b}"
    git restore --staged "$FILE"
    git restore "$FILE"
    continue
  fi

  git commit -m "$COMMIT_MSG" >/dev/null
  git push "$REMOTE" "HEAD:${b}"
  echo "  Updated ${b}"
done

git switch "$START_BRANCH" >/dev/null
echo "Done."
